#6: Como funciona a renderização

    

     

    



    <div id="react06"></div>
    <button id="btn06">stop</button> <button id="btn06start">start</button>

    <script type="text/babel">
        function mostrarHora06(){
            let elemento = <p className="tit"> {new Date().toLocaleTimeString()} </p>;
            
            let react06 = cerd('#react06');
            react06.render(elemento);
        }
        
        let react06_interval = setInterval(mostrarHora06, 1000);
        let btn06 = qs('#btn06');
        btn06.addEventListener('click', () => {
            clearInterval(react06_interval);
            
        });
        let btn06start = qs('#btn06start');
        btn06start.addEventListener('click', () => {
            react06_interval = setInterval(mostrarHora06, 1000);
        });
    </script>




#7: Introdução a Componentes e Props (1/2)

    Componentes: são itens independentes, separados.
    Exemplo: uma rede social, tem o feed com várias postagens,
    sendo cada postagem um componente, é idenpendente uma da outra,
    e ela funciona ou tem a máquina funcionando com ela.

        let elemento07 = (

    </script>
